Multiplots in Python

Creating a plot in Python is fairly easy. Here I show how you can create a multiplot with a for loop. To give you a background, the plot is made of spectra of the two stars that we have been looking at. There are 6 observations each of the stars. Here, I plot the H-alpha region. The script below creates a plot as shown below:

#-------------------------------------------
# IMPORT STATEMENTS
#-------------------------------------------
import numpy as np
import matplotlib.pyplot as plt
import glob

#-------------------------------------------
# Star 1: GETTING THE FILES TO CREATE THE PLOT 
#-------------------------------------------
path1 = '/Users/rohit/SOFTWARE/git/APG_EB/TARGETS/KIC_02445134/STITCHED_SPECTRA/'
files1 = glob.glob(path1+'*r_*_SPEC.txt')

#-------------------------------------------
# Star 2: GETTING THE FILES TO CREATE THE PLOT
#-------------------------------------------
path2 = '/Users/rohit/RESEARCH/ANALYSIS/TODCOR/KEPLER_EBS/HET/2m1937+3811/calspectra/STITCHED_SPECTRA/'
files2 = glob.glob(path2+'*r_*_SPEC.txt')

#-------------------------------------------
# BEGIN PLOTTING FIGURE 1 
#-------------------------------------------
fig = plt.figure()
#-------------------------------------------
# CREATE THE LEFT FIGURE FOR STAR 1 
#-------------------------------------------
ax1 = fig.add_axes([0.07, 0.09, 0.40, 0.89])
plt.minorticks_on()
plt.xlim(6555, 6575)
plt.ylabel('Normalized Flux + Constant')
plt.ylim(0.2, 4.4)
plt.xlabel('Wavelength (Angstroms)')
# Read in the file corresponding to the first observation. 
file1 = files1[0]
data = np.genfromtxt(file1, dtype=float, names="wl, fl", skiprows=1)
wl = data['wl']
fl = data['fl']
# Plot the first one
plt.plot(wl, fl, 'r-', color = '#38a996', label = 'KIC 02445134')
# Run the for loop through the files and plot every other observation. 
for i in range(1,len(files1)):
  this_file = files1[i]
  data1 = np.genfromtxt(this_file, dtype=float, names="wl, fl", skiprows=1)
  wl1 = data1['wl']
  fl1 = data1['fl']
  plt.plot(wl1, fl1+(i*0.6), linestyle='-', color = '#38a996')
plt.legend(numpoints=1, loc=2)
#-------------------------------------------
# CREATE THE RIGHT FIGURE FOR STAR 2 
#-------------------------------------------
ax1 = fig.add_axes([0.53, 0.09, 0.40, 0.89])
plt.minorticks_on()
plt.xlim(6550, 6570)
plt.ylim(0.3, 4.5)
plt.yticks([])
plt.xlabel('Wavelength (Angstroms)')
file2 = files2[0]
# Plot the first one
data = np.genfromtxt(file2, dtype=float, names="wl, fl", skiprows=1)
wl = data['wl']
fl = data['fl']
plt.plot(wl, fl, linestyle='-', color = '#eb8d89', label = 'KIC 03003991')
# Run the for loop through the files and plot every other observation. 
for j in range(1,len(files2)):
  this_file = files2[j]
  data2 = np.genfromtxt(this_file, dtype=float, names="wl, fl", skiprows=1)
  wl2 = data2['wl']
  fl2 = data2['fl']
  plt.plot(wl2, fl2+(j*0.6), linestyle='-', color = '#eb8d89')
plt.legend(numpoints=1, loc=2)
plt.savefig('Halpha_EBs.png', dpi = 300)
plt.show()
